  let condition1 = false;
  if (condition1) {
    resolve('Promise1 is resolved');
  } else {
    reject('Promise1 is rejected');
  }
}).then((message) => {
  console.log(message);
  let condition2 = true;
  if(condition2) {
    return 'Promise2 is resolved';

    return 'Promise2 is rejected';
  }
